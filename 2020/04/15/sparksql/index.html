利用反射推导schema
···java
package com.sqldemo
import org.apache.spark.sql.SparkSession
//方法一，利用反射，建立模式匹配类
//利用反射，rdd转换成dataFrame，模式匹配
case class Person(val name:String,val age:Int)
object sqldemo1 {
def main(args: Array[String]): Unit = {
//创建一个sparkSession对象
val sparkSession = SparkSession.builder().appName(“casesql”).master(“local[2]”).getOrCreate()
//创建SparkContext对象
val sc = sparkSession.sparkContext
//读取数据文件，获得一个rdd
val rdd1 = sc.textFile(“D:/people.txt”).map(.split(“,”))
//将rdd与样例类关联
val personRDD = rdd1.map(x=>Person(x(0),x(1).toInt))
//将personRDD转化成dataFrame，需导入隐转换
import sparkSession.implicits.
val personDF = personRDD.toDF()
//dataFrame风格操作
personDF.printSchema()
personDF.show()
personDF.select(“name”,”age”).show()
personDF.select($”age”>30).show()
//sql风格操作
personDF.createTempView(“person”)
sparkSession.sql(“select * from person”).show()
sparkSession.sql(“select * from person where age>30”).show()
//关闭sparkSession对象
sparkSession.stop()
}

}
